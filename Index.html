<script>

const video = document.getElementById('video');

const canvas = document.getElementById('canvas');

const ctx = canvas.getContext('2d');

let particles = [];

let handsData = [];

let theme = "cyan";

/* -------- KAMEHAMEHA STATE -------- */

let charging = false;

let chargePower = 0;

let beamActive = false;

let beamDir = {x:0,y:0};

const themes = {

  purple:["#fff","#a855f7"],

  gold:["#fff","#f59e0b"],

  cyan:["#fff","#06b6d4"]

};

function setTheme(t){ theme=t; }

function resize(){

  canvas.width = window.innerWidth;

  canvas.height = window.innerHeight;

}

window.onresize = resize;

resize();

navigator.mediaDevices.getUserMedia({video:true})

.then(stream=> video.srcObject = stream);

function getColor(i=0){

  if(theme==="rainbow") return `hsl(${i*18},100%,60%)`;

  return themes[theme][1];

}

/* -------- BASIC EFFECTS -------- */

function drawGlowLine(x1,y1,x2,y2,color){

  const layers=[

    {w:40,a:0.03},

    {w:25,a:0.08},

    {w:12,a:0.2},

    {w:4,a:1}

  ];

  layers.forEach(l=>{

    ctx.strokeStyle=color;

    ctx.globalAlpha=l.a;

    ctx.lineWidth=l.w;

    ctx.beginPath();

    ctx.moveTo(x1,y1);

    ctx.lineTo(x2,y2);

    ctx.stroke();

  });

  ctx.globalAlpha=1;

}

function drawParticles(){

  particles.forEach(p=>{

    p.x+=p.vx;

    p.y+=p.vy;

    p.life--;

    ctx.fillStyle=p.color;

    ctx.globalAlpha=p.life/40;

    ctx.beginPath();

    ctx.arc(p.x,p.y,2,0,Math.PI*2);

    ctx.fill();

  });

  ctx.globalAlpha=1;

  particles=particles.filter(p=>p.life>0);

}

/* -------- ENERGY BALL -------- */

function drawEnergyBall(x,y,r){

  const g=ctx.createRadialGradient(x,y,5,x,y,r);

  g.addColorStop(0,"white");

  g.addColorStop(1,getColor());

  ctx.fillStyle=g;

  ctx.beginPath();

  ctx.arc(x,y,r,0,Math.PI*2);

  ctx.fill();

  // aura particles

  for(let i=0;i<5;i++){

    particles.push({

      x:x+(Math.random()-0.5)*r,

      y:y+(Math.random()-0.5)*r,

      vx:(Math.random()-0.5)*2,

      vy:(Math.random()-0.5)*2,

      life:30,

      color:getColor()

    });

  }

}

/* -------- BEAM -------- */

function drawBeam(x,y,dir){

  const len = 2000;

  drawGlowLine(

    x,y,

    x+dir.x*len,

    y+dir.y*len,

    getColor()

  );

}

/* -------- MAIN DRAW -------- */

function draw(results){

  ctx.clearRect(0,0,canvas.width,canvas.height);

  ctx.save();

  ctx.scale(-1,1);

  ctx.drawImage(video,-canvas.width,0,canvas.width,canvas.height);

  ctx.restore();

  handsData = results.multiHandLandmarks || [];

  document.getElementById("status").innerText="Hands: "+handsData.length;

  /* ---- KAMEHAMEHA LOGIC ---- */

  if(handsData.length === 2){

    const h1 = handsData[0][9];

    const h2 = handsData[1][9];

    const x1 = h1.x*canvas.width;

    const y1 = h1.y*canvas.height;

    const x2 = h2.x*canvas.width;

    const y2 = h2.y*canvas.height;

    const midX = (x1+x2)/2;

    const midY = (y1+y2)/2;

    const dist = Math.hypot(x1-x2,y1-y2);

    // Start charging

    if(dist < 150){

      charging = true;

      chargePower += 0.02;

      if(chargePower > 1) chargePower = 1;

      drawEnergyBall(midX,midY,50 + chargePower*80);

    } else {

      // Release beam

      if(charging && chargePower > 0.3){

        beamActive = true;

        beamDir.x = (midX - canvas.width/2)/canvas.width;

        beamDir.y = (midY - canvas.height/2)/canvas.height;

      }

      charging = false;

      chargePower *= 0.95;

    }

    // Draw beam

    if(beamActive){

      drawBeam(midX,midY,beamDir);

      chargePower -= 0.02;

      if(chargePower <= 0){

        beamActive = false;

      }

    }

  } else {

    charging = false;

    beamActive = false;

    chargePower *= 0.9;

  }

  drawParticles();

}

/* -------- INIT -------- */

const hands = new Hands({

  locateFile: f=>`https://cdn.jsdelivr.net/npm/@mediapipe/hands/${f}`

});

hands.setOptions({

  maxNumHands:2,

  modelComplexity:1,

  minDetectionConfidence:0.7,

  minTrackingConfidence:0.5

});

hands.onResults(draw);

const camera = new Camera(video,{

  onFrame: async ()=>{

    await hands.send({image:video});

  },

  width:1280,

  height:720

});

camera.start();

setTimeout(()=>{

  document.getElementById("loading").style.display="none";

},2000);

</script>
